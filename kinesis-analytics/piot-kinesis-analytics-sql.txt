CREATE OR REPLACE STREAM "DESTINATION_SQL_STREAM_001"  ("empty_count" INT);
CREATE OR REPLACE PUMP "STREAM_PUMP" AS INSERT INTO "DESTINATION_SQL_STREAM_001" 
SELECT STREAM COUNT("empty") OVER SLIDING_WINDOW AS "empty_count" FROM "SOURCE_SQL_STREAM_001"
WINDOW SLIDING_WINDOW AS (PARTITION BY PARTITION_KEY RANGE INTERVAL '1' MINUTE PRECEDING);
